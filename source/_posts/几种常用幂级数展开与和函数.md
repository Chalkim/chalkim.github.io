### 常见函数的级数展开



$$
\begin{equation}
\begin{split}
\frac{1}{1-x} = \sum_{n=0}^{\infty} x^n
\end{split}
\end{equation}
$$

$$
\begin{equation}
\begin{split}
\frac{1}{1+x} = \sum_{n=0}^{\infty} (-1)^nx^n
\end{split}
\end{equation}
$$

$$
\begin{equation}
\begin{split}
e^x = \sum_{n=0}^{\infty} \frac{x^n}{n!}
\end{split}
\end{equation}
$$

$$
\begin{equation}
\begin{split}
\sin x = \sum_{n=0}^{\infty} \frac{(-1)^nx^{2n+1}}{(2n+1)!}
\end{split}
\end{equation}
$$

$$
\begin{equation}
\begin{split}
\cos x = \sum_{n=0}^{\infty} \frac{(-1)^nx^{2n}}{(2n)!}
\end{split}
\end{equation}
$$

$$
\begin{equation}
\begin{split}
\ln(1+x) = \sum_{n=0}^{\infty} \frac{(-1)^{n-1}x^{n}}{n}
\end{split}
\end{equation}
$$

$$
\begin{equation}
\begin{split}
\arctan x&=\sum\limits_{n=0}^\infty\frac{(-1)^nx^{2n+1}}{2n+1}
\end{split}
\end{equation}
$$

<details>
<summary>$f(x)=\arctan x$</summary>
$$
\begin{aligned}
f'(x)&=\frac{1}{1+x^2}=\sum\limits_{n=0}^\infty{(-1)}^nx^{2n}\\
f(x) &= \int_{0}^{x}{f'(x)}\mathrm{d}x + f(0)\\
&=\sum\limits_{n=0}^\infty\frac{(-1)^nx^{2n+1}}{2n+1}
\end{aligned}
$$
</details>
